# =============================================================================
# MarkFlow 环境变量配置示例
# 使用方式：复制为项目根目录 .env 或 .env.local（.local 可不入库）
# 注意：pnpm dev 使用 `vite --mode test`，会合并 .env + .env.test，后者覆盖同名项
#       pnpm build / build:test 等使用对应 mode 的 .env.*
# =============================================================================

# -----------------------------------------------------------------------------
# 应用基础信息
# -----------------------------------------------------------------------------

VITE_APP_TITLE=MarkFlow
VITE_APP_DESC=Markdown Workspace

# 部署基础路径，子目录必须以 / 结尾
VITE_BASE_URL=/markflow/

# -----------------------------------------------------------------------------
# 路由与鉴权
# -----------------------------------------------------------------------------

# Vue Router：history | hash | memory
VITE_ROUTER_HISTORY_MODE=history

# static = 前端生成路由（对接 ai-server 推荐）
# dynamic = 后端下发路由与侧栏（需 ai-server menus/user-routes）
VITE_AUTH_ROUTE_MODE=dynamic

# Y = 无后端时跳过登录；对接 ai-server 请 N
VITE_DEV_BYPASS_AUTH=N

# static 模式登录后首页路由 key
VITE_ROUTE_HOME=home

# -----------------------------------------------------------------------------
# 图标
# -----------------------------------------------------------------------------

VITE_ICON_PREFIX=icon
VITE_ICON_LOCAL_PREFIX=local-icon
VITE_MENU_ICON=mdi:menu

# -----------------------------------------------------------------------------
# 后端服务
# -----------------------------------------------------------------------------

# 主后端地址（ai-server 默认 3000，API 前缀 /api）
VITE_SERVICE_BASE_URL=http://localhost:3000

# 其它独立后端（JSON 对象，key 对应代码里 otherBaseURL 的键，如 demo）
# 示例：{"demo":"http://localhost:4000"}
VITE_OTHER_SERVICE_BASE_URL={"demo":"http://localhost:4000"}

# 开发代理（仅 dev）
VITE_HTTP_PROXY=Y
VITE_PROXY_LOG=Y

# -----------------------------------------------------------------------------
# 业务状态码（仅配置响应体 code，勿填 HTTP 401/403）
# 详见：docs/环境变量配置说明.md
# -----------------------------------------------------------------------------

# 成功：200 通用成功，201 创建成功
VITE_SERVICE_SUCCESS_CODE=200,201

# Token 过期可刷新 → 调 refresh 接口后重试
# 1202 TOKEN_EXPIRED
VITE_SERVICE_EXPIRED_TOKEN_CODES=1202

# 已登录立即登出（不含登录页 1200/1201）
# 1203 TOKEN_INVALID | 1204 TOKEN_REVOKED | 1205 SESSION_EXPIRED
VITE_SERVICE_LOGOUT_CODES=1203,1204,1205

# 弹窗确认后登出
# 1206 ACCOUNT_LOCKED | 1207 ACCOUNT_DISABLED | 2000 ACCOUNT_BLACKLISTED
VITE_SERVICE_MODAL_LOGOUT_CODES=1206,1207,2000

# -----------------------------------------------------------------------------
# 构建与运行
# -----------------------------------------------------------------------------

# GitHub Pages 等静态演示：不请求远程 API，注入演示账号
# VITE_STATIC_DEMO=Y

VITE_SOURCE_MAP=N
VITE_DROP_CONSOLE=Y
VITE_STORAGE_PREFIX=admin_
VITE_AUTOMATICALLY_DETECT_UPDATE=Y

# 可选
# VITE_ICONIFY_URL=https://api.iconify.design
# VITE_DEVTOOLS_LAUNCH_EDITOR=code
